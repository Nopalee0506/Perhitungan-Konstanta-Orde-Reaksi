import streamlit as st import math t = [0,1,2,3] A = [10,7,5,3.6] A0 = A[0] y1 = [math.log(A0/a) for a in A] y2 = [(1/a - 1/A0) for a in A] def regresi(x,y): n=len(x) sx=sum(x); sy=sum(y) sxy=sum(x[i]*y[i] for i in range(n)) sx2=sum(xi*xi for xi in x) m=(n*sxy-sx*sy)/(n*sx2-sx*sx) return m print("k Orde 1:", regresi(t,y1)) print("k Orde 2:", regresi(t,y2))
